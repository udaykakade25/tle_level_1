create table student (
  roll varchar(10) primary key,
  name varchar(50),
  attendance number,
  status varchar(20)
);
/
insert into student values
('s01','arun',65,null),
('s02','meera',80,null),
('s03','gopal',90,null),
('s04','maya',55,null),
('s05','anita',75,null);


declare
  v_roll student.roll%type := '&roll';
  v_att student.attendance%type;
begin
  select attendance into v_att from student where roll = v_roll;
  if v_att < 75 then
    update student set status='detained' where roll=v_roll;
    dbms_output.put_line('term not granted');
  else
    update student set status='not detained' where roll=v_roll;
    dbms_output.put_line('term granted');
  end if;
exception
  when no_data_found then
    dbms_output.put_line('student not found');
end;
/
